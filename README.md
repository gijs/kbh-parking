# utility

    sinh = (x) -> (Math.pow(Math.E,x) - Math.pow(Math.E, -x))/2
    
    tile2coordZoom = (zoom) -> (x, y) ->
      scale = Math.pow 0.5, zoom
      return [
        x * scale * 360.0 - 180.0
        180 / Math.PI * Math.atan sinh(Math.PI * (1 - 2 * y * scale))
      ]
